Aftab / آفتاب

A research benchmark for encoder design in deep RL.

Standardized training, evaluation, and reproducibility tools for comparing CNN-based encoders in PQN across Atari.

from aftab import Aftab
from aftab import aftab_environments

for environment in aftab_environments:
    agent = Aftab(encoder="gamma", frames="pilot")
    agent.train(environment=environment, seed=1)
    agent.log()
